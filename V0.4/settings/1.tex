\makeatletter
\def\cleardoublepage{
  \clearpage
  \if@twoside
    \ifodd
      \c@page
    \else
      \begingroup
        \mbox{}
        \thispagestyle{empty}
        \newpage
        \if@twocolumn\mbox{}\newpage\fi
      \endgroup
    \fi
  \fi}
